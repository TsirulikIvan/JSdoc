/**
Возвращает площадь треугольника

@param {shall} a - Первая сторона треугольника
@param {shall} b - Вторая сторона треугольника
@param {shall} c - Третья сторона треугольника


*/
function geron (a,b,c){
	let pp = (a + b + c) / 2;
	return Math.sqrt(pp*(pp-a)*(pp-b)*(pp-c));
}
console.log("S = " + geron(3,4,5));
